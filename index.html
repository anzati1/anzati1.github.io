NUT to wOS ALCS

Realm: SERVER
Description: Converts NUT Items to wOS ALCS Items.
Status: Working

function ConvertToNut()
if (not nut) then
Error("nutscript was not found")

return
end

for k, v in pairs(wOS.ItemList) do
nut.item.list[k] = nil
nut.item.Register(k, nil, nil, nil, true)
local item = nut.item.list[k]
item.WOS = v -- config wiltos
item.name = v.Name
item.model = Model(v.Model)

if (v.Icon or v.icon) then
item.icon = v.Icon
end

item.noBusiness = false
item.description = v.Description
item.category = "ALCS Crystals"
item.width = 1
item.height = 1
item.class = "nut_loot"
end

print("WOS Items loaded in NUTSCRIPT")
end

hook.Add("wOS.ALCS.PostLoaded", "ConvertToNUTT", function()
ConvertToNut()
end)

local multiplier = 0
local wepp = FindMetaTable("Weapon")

function wepp:DoLightV2()
local stance = self:GetStance()
local form = wOS.Form.LocalizedForms[self:GetForm()]

if not form then
print("[wOS] Invalid form! Animations will not operate properly!")
self:SetDelay(0)

return
end

local formdata = wOS.Form.Singles[form][stance]
local movedata = formdata["light_forward" .. math.random(1, formdata["amt_forward"])]
local time

if self.Owner:KeyDown(IN_MOVELEFT) then
movedata = formdata["light_left" .. math.random(1, formdata["amt_left"])]
elseif self.Owner:KeyDown(IN_MOVERIGHT) then
movedata = formdata["light_right" .. math.random(1, formdata["amt_right"])]
end

if not movedata then
print("[wOS] Invalid move data for form: " .. form .. "!")
self:SetDelay(0)

return
end

if not self.Owner:GetSequence() == formdata["idle"] then return end
local time = self.Owner:SetSequenceOverride(movedata["Sequence"], movedata["Time"], movedata["Rate"]) or 0.5

if not self.CanMoveWhileAttacking then
self.Owner:SetNWFloat("wOS.SaberAttackDelay", CurTime() + time)
end

multiplier = multiplier + 1
time = time or 0.5
self:SetDelay(time)
self.AttackTime = CurTime() + time
print(multiplier)
local time2 = movedata["Time"] or 0.5
local rate2 = movedata["Rate"] or 1
PrintMessage(HUD_PRINTCENTER, multiplier .. " | " .. movedata["Sequence"] .. " | " .. time2 .. " | " .. rate2)

timer.Simple((time2 * formdata["amt_forward"]), function()
multiplier = 1
end)
end
